class nemico

public:
	void attacca(int& play_vita) {
		cout << "ti colpisce" << endl;
		bool crittato = rand() % 100 < critn;
		if (crittato)
		{
			cout << "colpo critico" << endl;
			play_vita -= dannon * 2;
		}
		else
			play_vita -= dannon;
		if (play_vita < 0)
			play_vita = 0;
	}
	nemico(string c, int v, int d, int s, int cr, vector<string>sprit, vector<string>sprit_attk, vector<string>sprit_kill)

		classen = c;
		vitan = v;

		critn = cr;
		sprite = sprit;
		sprite_attk = sprit_attk;
		sprite_kill = sprit_kill;

	void pos_nemico1(vector<maps*>word)
	{
		wordn = rand() % word.size();
		posxn = rand() % word[wordn]->getmapxmax();
		posyn = rand() % word[wordn]->getmapymax();
	}
	string getclasse() const { return classen; }
	int getdanno() const { return dannon; }
	int getvita() const { return vitan; }

	int getspeed() const { return speedn; }
	int& getposnx() { return posxn; }

void player::turno(int& mob_vita, int danno_nemico, const nemico& mob, const player& gamer)
{
	cout << hp << " hp" << endl;
	char scelta = _getch();
	while (scelta != '1' && scelta != '2')
	{
		cout << "inserire valore valido" << endl;
		cin >> scelta;
	}
	if (scelta == '1')
	{
		drawmach(mob.getSprite(), gamer.getSpriteAttk(), mob, gamer);
		attacchi(mob_vita);
		wait(1500);
	}

		cura(danno_nemico);
}

bool prima(vector<nemico*> mob, player giocatore, int index)
{
	bool primo;
	if (mob[index]->getspeed() < giocatore.getspeed())
		primo = true;
	else
		primo = false;
	return primo;
}

void drawmach(vector<string>spritenem, vector<string>spriteplay, const nemico& mob, const player& gamer)

}

void combattimento(vector<nemico*>& mob, player& giocatore, int index, int& kill)
{
	cout << "\033[2J\033[1;1H";
	this_thread::sleep_for(chrono::milliseconds(100));
	cout << "e comparso un... " << mob[index]->getclasse();
	this_thread::sleep_for(chrono::milliseconds(1200));
	drawmach(mob[index]->getSprite(), giocatore.getSprite(), *mob[index], giocatore);
	while (true)
	{
		drawmach(mob[index]->getSprite(), giocatore.getSprite(), *mob[index], giocatore);
		this_thread::sleep_for(chrono::milliseconds(1000));
		if (prima(mob, giocatore, index))
		{
			giocatore.turno(mob[index]->getvita(), mob[index]->getdanno(), *mob[index],giocatore);
			if (mob[index]->getvita() > 0)
			{
				drawmach(mob[index]->getSpriteAttk(), giocatore.getSprite(), *mob[index], giocatore);
				mob[index]->attacca(giocatore.getvita());
			}
		}
		else
		{
			drawmach(mob[index]->getSpriteAttk(), giocatore.getSprite(), *mob[index], giocatore);
			mob[index]->attacca(giocatore.getvita());
			this_thread::sleep_for(chrono::milliseconds(1500));
			if (giocatore.getvita() > 0)
			{
				giocatore.turno(mob[index]->getvita(), mob[index]->getdanno(), *mob[index],giocatore);
			}
		}
		if (mob[index]->getvita() <= 0)
		{
			drawmach(mob[index]->getSpriteKill(),giocatore.getSprite(),*mob[index],giocatore);
			delete mob[index];
			mob.erase(mob.begin() + index);
			kill += 1;
			this_thread::sleep_for(chrono::milliseconds(1500));
			break;
		}
		this_thread::sleep_for(chrono::milliseconds(1500));
		if (giocatore.getvita() <= 0)
		{
			drawmach(mob[index]->getSprite(), giocatore.getSpriteKill(), *mob[index], giocatore);
			break;
		}
	}

}

void cambiamappa(player& gamer, vector<maps*>& mondo)
{
	int y = gamer.getposy();
	int x = gamer.getposx();
	int ymax = mondo[gamer.getmap()]->getmapymax() - 1;
	int xmax = mondo[gamer.getmap()]->getmapxmax() - 1;

	if (gamer.getmap() == 0)
	{
		if (x >= xmax)  
		{
			gamer.getmap() = 1;
			gamer.getposx() = 1;  
		}
		else if (y <= 0)  
		{
			gamer.getmap() = 2;
			gamer.getposy() = ymax-1;
		}
	}
	else if (gamer.getmap() == 1)
	{
		if (x <= 0)  
		{
			gamer.getmap() = 0;
			gamer.getposx() = xmax - 1;
		}
	}
	else if (gamer.getmap() == 2)
	{
		if (y >= ymax)  
		{
			gamer.getmap() = 0;
			gamer.getposy() = 1;
		}
	}
}

		int newposy = you.getposy();
		int newposx = you.getposx();
		if (input == 'w')
			newposy--;
		if (input == 'a')
			newposx--;
		if (input == 's')
			newposy++;
		if (input == 'd')
			newposx++;
		if (mondo[you.getmap()]->gettile(newposy,newposx) != '#')
		{
			for (int i = 0; i < mob.size(); i++)
			{
				int ym = mob[i]->getposny();
				int xm = mob[i]->getposnx();
				if (you.getmap()==mob[i]->getword() && newposy == ym && newposx == xm)
				{
					combattimento(mob, you, i, kill);
					break;
				}
				
			}
			if (you.getvita() > 0)
			{
				you.getposy() = newposy;
				you.getposx() = newposx;
				cambiamappa(you, mondo);
			}
			else
				break;
		}
		if (you.getvita() <= 0)
			break;
		this_thread::sleep_for(chrono::milliseconds(300));
		if (mob.size() == 0)
		{
			nmob = rand() % 8;
			for (int i = 0; i <= nmob; i++)
			{
				crea_mob(mob);
			}
			for (int i = 0; i < mob.size(); i++)
			{
				do { mob[i]->pos_nemico1(mondo); }
				while (mondo[mob[i]->getword()]->gettile(mob[i]->getposny(),mob[i]->getposnx()) == '#' ||  //controlla pos nemico diversa da muri
					  (you.getmap()==mob[i]->getword() && you.getposy() == mob[i]->getposny() && you.getposx() == mob[i]->getposnx()));//controlla pos nemico diversa da player

			}
		}
	}
	cout << "game over";
	cin.ignore();
	cin.get();
	return 0;
